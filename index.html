# This chart shows how Netflix content is rated. 
# Each rating tells us what age group the content is suitable for:
# TV-MA: For mature adults (18+)
# TV-14: May not be suitable for children under 14
# TV-PG: Parental guidance suggested
# R: Restricted (Under 17 requires accompanying adult)
# PG: Parental guidance
# G: General audience, suitable for all ages
# TV-G: Suitable for all ages
# NC-17: Adults only (18+)

df['rating'].dropna().value_counts().plot(kind='bar' , color='pink')
plt.title('Distribution of content ratings on netflix')
plt.xlabel('Rating')
plt.ylabel('Number of Titles')
plt.xticks(rotation=45)
plt.grid(axis='y')
plt.show()
